	\answer{
		\begin{proof}
			\begin{itemize}
				\item $a \sim b \implies b - a \in \bZ \implies a - b \in \bZ \implies b \sim a$. Verifying, symmetricity. 
				\item $a - a = 0 \in \bZ \implies a \sim a$. Verifying reflexivity. 
				\item If $a \sim b \implies b-a \in \bZ$ and $b \sim c \implies c-b \in \bZ$, then adding those two relations we get $a - c \in \bZ \implies c \sim a$. 
			\end{itemize}
	Thus, $\sim$ on $\bR$ is indeed an equivalence relation. 



		\end{proof}
	}
